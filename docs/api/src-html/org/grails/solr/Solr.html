024    package org.grails.solr;
025    import java.lang.annotation.*;
026    
027    /**
028     * Describes the Solr annotated API element.
029     */
030    @Retention(RetentionPolicy.RUNTIME)
031    @Target(ElementType.FIELD)
032    public @interface Solr {
033        String field() default "";
034                    boolean asText() default false;;
035    }
